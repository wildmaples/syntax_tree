def run(item)
  handler = item.handler

  warning = "[#{Color.yellow("warn")}] #{item.filepath}"
  formatted = handler.format(item.source)

  raise NonIdempotentFormatError if formatted != handler.format(formatted)
rescue StandardError
  warn(warning)
  raise
end
