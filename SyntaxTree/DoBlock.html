    class SyntaxTree::DoBlock

    DoBlock represents passing a block to a method call using the do and end keywords.

    method do |value|
end

def pretty_print(q)
  q.group(2, "(", ")") do
    q.text("do_block")

    if block_var
      q.breakable
      q.pp(block_var)
    end

    q.breakable
    q.pp(bodystmt)

    q.pp(Comment::List.new(comments))
  end
end
